import React, { useContext } from 'react';
import { SharedStateContext } from './SharedStateContext';

const MeuComponente = () => {
  const {
    initialY,
    speed,
    arrastando2
  } = useContext(SharedStateContext);

  const handleSomething = () => {
    initialY.current = 100;
    arrastando2.current = true;
    speed.current = 1.2;
    console.log('initialY:', initialY.current);
  };

  return <button onClick={handleSomething}>Clique</button>;
};
